pipeline {
    // Run this pipeline on any available Jenkins agent
    agent any

    tools {
        // Maven tool installation configured in Jenkins as "M3"
        maven "M3"
    }

    environment {
        // Demo/local repo path inside Jenkins container
        REPO_PATH      = "/repos/test/.git"
        REPO_BRANCH    = "main"

        // Minecraft deploy target (mounted volume path visible to Jenkins container)
        MC_PLUGIN_DIR  = "/mc-data/plugins"
        MC_PLUGIN_JAR  = "DemoPlugin.jar"

        // RCON connection details for the Minecraft server container
        RCON_HOST      = "mc-dev"
        RCON_PORT      = "25575"

        // Redmine connection details for posting build status notes
        REDMINE_URL    = "http://redmine:3000"
        REDMINE_ISSUE  = "3"
    }

    stages {
        stage('Checkout Demo Plugin') {
            steps {
                // Clean workspace, clone the demo plugin repository, and check out the branch
                sh '''
                    set -e
                    rm -rf ./* ./.git ./.gitignore || true
                    git clone "$REPO_PATH" .
                    git checkout "$REPO_BRANCH"
                '''
            }
        }

        stage('Build and Test') {
            steps {
                // Build the plugin JAR with Maven and run tests (if any)
                sh '''
                    set -e
                    mvn -B -DskipTests=false -Dmaven.test.failure.ignore=true clean test package
                '''
            }
            post {
                always {
                    // Publish JUnit test reports if present.
                    // allowEmptyResults avoids failing/noisy warnings when no tests exist yet.
                    junit testResults: 'target/surefire-reports/*.xml', allowEmptyResults: true
                }
            }
        }

        stage('Archive Plugin Artifact') {
            steps {
                // Archive built JAR in Jenkins and fingerprint for traceability
                archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
            }
        }

        stage('Deploy to Minecraft Dev Server (RCON)') {
            steps {
                // Inject the RCON password securely from Jenkins credentials
                withCredentials([
                    string(credentialsId: 'mc-rcon-password', variable: 'RCON_PASS')
                ]) {
                    sh '''
                        set -e

                        # Install/build mcrcon locally in the workspace to communicate with the server
                        if ! command -v mcrcon >/dev/null 2>&1; then
                          echo "[CI] mcrcon not found. Building local copy..."

                          # check for the required build tools
                          if command -v apt-get >/dev/null 2>&1; then
                            apt-get update
                            DEBIAN_FRONTEND=noninteractive apt-get install -y git gcc make libc6-dev
                          # Alpine Linux fallback
                          elif command -v apk >/dev/null 2>&1; then
                            apk add --no-cache git build-base
                          else
                            echo "no supported package manager found."
                            exit 1
                          fi

                          # Clone mcrcon source into a temporary workspace directory
                          rm -rf .mcrcon-src
                          git clone https://github.com/Tiiffi/mcrcon.git .mcrcon-src

                          # Build the mcrcon binary from source
                          gcc -O2 -o .mcrcon-src/mcrcon .mcrcon-src/mcrcon.c

                          # Make binary executable and prepend to PATH for this shell
                          chmod +x .mcrcon-src/mcrcon
                          export PATH="$PWD/.mcrcon-src:$PATH"
                        fi

                        # Re-check mcrcon availability
                        if ! command -v mcrcon >/dev/null 2>&1; then
                          echo "ERROR: mcrcon is still unavailable after install/build attempt"
                          exit 1
                        fi

                        # Find the built plugin JAR (first match in target/)
                        JAR="$(ls -1 target/*.jar | head -n 1)"
                        [ -n "$JAR" ] && [ -f "$JAR" ]

                        # Copy plugin JAR into the locally mounted Minecraft plugins directory
                        mkdir -p "$MC_PLUGIN_DIR"
                        cp "$JAR" "$MC_PLUGIN_DIR/$MC_PLUGIN_JAR"

                        # Gracefully stop the server so the containers restart policy starts it again and reloads the plugin
                        mcrcon -H "$RCON_HOST" -P "$RCON_PORT" -p "$RCON_PASS" "stop"
                    '''
                }
            }
        }
    }

    post {
        success {
            // On successful build, post a success note to the linked Redmine issue
            withCredentials([string(credentialsId: 'redmine_api_key', variable: 'REDMINE_KEY')]) {
                sh '''
                    set -e

                    # Build JSON payload as a file to avoid quoting issues in shell/curl
                    cat > redmine-payload.json <<EOF
{"issue":{"notes":"(Jenkins) Plugin build SUCCESS.\\\\nBuild: ${BUILD_URL}\\\\nResult: SUCCESS"}}
EOF

                    # Update Redmine issue with the build result note
                    curl -sS -X PUT \
                      -H "X-Redmine-API-Key: $REDMINE_KEY" \
                      -H "Content-Type: application/json" \
                      --data @redmine-payload.json \
                      "$REDMINE_URL/issues/$REDMINE_ISSUE.json"
                '''
            }
        }

        failure {
            // On failed build, post a failure note to the linked Redmine issue
            withCredentials([string(credentialsId: 'redmine_api_key', variable: 'REDMINE_KEY')]) {
                sh '''
                    set -e

                    # Build JSON payload for failure status
                    cat > redmine-payload.json <<EOF
{"issue":{"notes":"(Jenkins) Plugin build FAILED.\\\\nBuild: ${BUILD_URL}\\\\nResult: FAILURE"}}
EOF

                    # Update Redmine issue with failure note
                    curl -sS -X PUT \
                      -H "X-Redmine-API-Key: $REDMINE_KEY" \
                      -H "Content-Type: application/json" \
                      --data @redmine-payload.json \
                      "$REDMINE_URL/issues/$REDMINE_ISSUE.json"
                '''
            }
        }

        unstable {
            // On unstable build (e.g., test issues), post an unstable note to Redmine
            withCredentials([string(credentialsId: 'redmine_api_key', variable: 'REDMINE_KEY')]) {
                sh '''
                    set -e

                    # Build JSON payload for unstable status
                    cat > redmine-payload.json <<EOF
{"issue":{"notes":"(Jenkins) Plugin build UNSTABLE.\\\\nBuild: ${BUILD_URL}\\\\nResult: UNSTABLE"}}
EOF

                    # Update Redmine issue with unstable note
                    curl -sS -X PUT \
                      -H "X-Redmine-API-Key: $REDMINE_KEY" \
                      -H "Content-Type: application/json" \
                      --data @redmine-payload.json \
                      "$REDMINE_URL/issues/$REDMINE_ISSUE.json"
                '''
            }
        }

        always {
            // clean the Jenkins workspace after the pipeline finishes
            cleanWs(deleteDirs: true, disableDeferredWipeout: true)
        }
    }
}
